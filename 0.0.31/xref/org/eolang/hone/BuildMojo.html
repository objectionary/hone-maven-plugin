5   package org.eolang.hone;
6   
7   import java.io.IOException;
8   import org.apache.maven.plugins.annotations.LifecyclePhase;
9   import org.apache.maven.plugins.annotations.Mojo;
10  import org.cactoos.Scalar;
11  import org.cactoos.io.OutputTo;
12  import org.cactoos.io.ResourceOf;
13  import org.cactoos.io.TeeInput;
14  import org.cactoos.scalar.IoChecked;
15  import org.cactoos.scalar.LengthOf;
16  import org.cactoos.scalar.Retry;
17  
18  /**
19   * Build Docker image.
20   *
21   * <p>This goal must be used only if you want to build a local custom
22   * Docker image for your project. This may be useful when you don't have
23   * network access to Docker Hub. In most cases, you have it and that's
24   * why don't need this goal.
25   * Instead, just use the <tt>pull</tt> goal, which will simply pull
26   * a required Docker image from the Hub.</p>
27   *
28   * <p>This goal is mostly for testing and CI/CD.</p>
29   *
30   * @since 0.1.0
31   */
32  @Mojo(name = "build", defaultPhase = LifecyclePhase.GENERATE_RESOURCES)
33  public final class BuildMojo extends AbstractMojo {
34  
35      @Override
36      public void exec() throws IOException {
37          try (Mktemp temp = new Mktemp()) {
38              final String[] files = {
39                  "Dockerfile", "entry.sh", "in-docker-pom.xml", "normalize.sh",
40                  "install-cabal.sh", "install-maven.sh",
41                  "install-phino.sh", "extensions.xml",
42              };
43              for (final String file : files) {
44                  new IoChecked<>(
45                      new LengthOf(
46                          new TeeInput(
47                              new ResourceOf(String.format("org/eolang/hone/scaffolding/%s", file)),
48                              new OutputTo(temp.path().resolve(file))
49                          )
50                      )
51                  ).value();
52              }
53              new Rules().copyTo(temp.path().resolve("rules"));
54              for (final String file : new String[] {"entry.sh", "normalize.sh"}) {
55                  temp.path().resolve(file).toFile().setExecutable(true);
56              }
57              this.timings.through(
58                  "build",
59                  () -> new IoChecked<>(
60                      new Retry<>(
61                          (Scalar<Object>) () -> new Docker(this.sudo).exec(
62                              "build",
63                              "--pull",
64                              "--progress=plain",
65                              "--tag", this.image,
66                              temp.path().toString()
67                          )
68                      )
69                  ).value()
70              );
71          }
72      }
73  }
