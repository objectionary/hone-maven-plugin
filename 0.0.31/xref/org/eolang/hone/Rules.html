5   package org.eolang.hone;
6   
7   import com.jcabi.log.Logger;
8   import java.io.IOException;
9   import java.nio.file.Path;
10  import java.util.Collection;
11  import java.util.HashMap;
12  import java.util.LinkedList;
13  import java.util.Map;
14  import java.util.regex.Pattern;
15  import org.cactoos.io.OutputTo;
16  import org.cactoos.io.ResourceOf;
17  import org.cactoos.io.TeeInput;
18  import org.cactoos.iterable.Mapped;
19  import org.cactoos.scalar.IoChecked;
20  import org.cactoos.scalar.LengthOf;
21  
22  /**
23   * Collection of YAML rules for Phino.
24   *
25   * @since 0.1.0
26   */
27  final class Rules {
28  
29      /**
30       * All of them.
31       */
32      private static final String[] ALL = {
33          "none", "thirty-three",
34      };
35  
36      /**
37       * Patterns.
38       */
39      private final Map<Pattern, Boolean> patterns;
40  
41      /**
42       * Ctor.
43       */
44      Rules() {
45          this("*");
46      }
47  
48      /**
49       * Ctor.
50       * @param ptns The pattern (comma-separated)
51       */
52      Rules(final String ptns) {
53          this.patterns = Rules.regexs(ptns);
54      }
55  
56      @Override
57      public String toString() {
58          return String.join(
59              " ",
60              new Mapped<>(
61                  t -> t.toString(),
62                  this.patterns
63              )
64          );
65      }
66  
67      /**
68       * Create a space-separated list of rule
69       * file names, for example, the rules defined by the user as
70       * "n*,thirty-three" this method will produce "none.yml thirty-three.yml".
71       * @return List of rule YML files separated by space
72       */
73      public Iterable<String> yamls() {
74          final Collection<String> files = new LinkedList<>();
75          for (final String name : Rules.ALL) {
76              if (!this.matches(name)) {
77                  continue;
78              }
79              files.add(String.format("%s.yml", name));
80          }
81          return files;
82      }
83  
84      /**
85       * Copy them all to a destination directory.
86       * @param dir Destination directory to copy to
87       * @throws IOException If fails
88       */
89      public void copyTo(final Path dir) throws IOException {
90          if (dir.toFile().getParentFile().mkdirs()) {
91              Logger.debug(this, "Directory created at %[file]s", dir);
92          }
93          for (final String file : this.yamls()) {
94              final Path target = dir.resolve(file);
95              new IoChecked<>(
96                  new LengthOf(
97                      new TeeInput(
98                          new ResourceOf(
99                              String.format("org/eolang/hone/rules/%s", file)
100                         ),
101                         new OutputTo(target)
102                     )
103                 )
104             ).value();
105         }
106     }
107 
108     /**
109      * This rule name is good?
110      * @param name THe name of the rule, e.g. "thirty-three"
111      * @return TRUE if it's good
112      */
113     private boolean matches(final CharSequence name) {
114         boolean matches = false;
115         for (final Map.Entry<Pattern, Boolean> ent : this.patterns.entrySet()) {
116             if (ent.getKey().matcher(name).matches()) {
117                 if (ent.getValue()) {
118                     matches = true;
119                 }
120                 break;
121             }
122         }
123         return matches;
124     }
125 
126     /**
127      * Convert list of patterns to regular expressions.
128      * @param ptns List of them
129      * @return Map of regular expression patterns
130      */
131     private static Map<Pattern, Boolean> regexs(final String ptns) {
132         final Map<Pattern, Boolean> list = new HashMap<>(0);
133         for (final String ptn : ptns.split("\\s*,\\s*")) {
134             if (ptn.isEmpty()) {
135                 continue;
136             }
137             boolean negative = false;
138             String body = ptn;
139             if (ptn.charAt(0) == '!') {
140                 negative = true;
141                 body = body.substring(1);
142             }
143             list.put(
144                 Pattern.compile(
145                     String.format(
146                         "^\\Q%s\\E$",
147                         body.replace("*", "\\E.*\\Q")
148                     )
149                 ),
150                 !negative
151             );
152         }
153         return list;
154     }
155 }
