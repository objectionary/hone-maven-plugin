Hone Maven Plugin

This README documentation explains it in details.

Use it like this:

<build>
  <plugins>
    <plugin>
      <groupId>org.eolang</groupId>
      <artifactId>hone-maven-plugin</artifactId>
      <version>0.0.35</version>
      <executions>
        <execution>
          <phase>process-classes</phase>
          <goals>
            <goal>optimize</goal>
          </goals>
        </execution>
      </executions>
    </plugin>
  </plugins>
</build>
