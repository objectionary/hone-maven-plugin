5   package org.eolang.hone;
6   
7   import java.io.IOException;
8   import org.junit.jupiter.api.extension.ConditionEvaluationResult;
9   import org.junit.jupiter.api.extension.ExecutionCondition;
10  import org.junit.jupiter.api.extension.ExtensionContext;
11  
12  /**
13   * This annotation disables JUnit tests when Docker is
14   * not runnable in command line.
15   *
16   * @since 0.1.0
17   */
18  public final class DisabledWithoutDockerCondition implements ExecutionCondition {
19      @Override
20      public ConditionEvaluationResult evaluateExecutionCondition(final ExtensionContext ctx) {
21          ConditionEvaluationResult result;
22          try {
23              new Docker().exec("ps");
24              result = ConditionEvaluationResult.enabled("Docker is available");
25          } catch (final IOException ex) {
26              result = ConditionEvaluationResult.disabled("Docker is not available");
27          }
28          return result;
29      }
30  }
