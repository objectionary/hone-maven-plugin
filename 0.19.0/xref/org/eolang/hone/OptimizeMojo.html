32  /**
33   * Converts Bytecode to Bytecode in order to make it faster.
34   *
35   * <p>This goal takes every {@code .class} file from the
36   * <tt>target/classes/</tt> directory, converts it to {@code .xmir}
37   * format (which is XML representation of <a href="https://www.eolang.org">EO</a>),
38   * then converts {@code .xmir} to {@code .phi} (which is
39   * <a href="https://arxiv.org/abs/2111.13384">ùúë-calculus</a>),
40   * then optimizes it via
41   * <a href="https://github.com/objectionary/phino">phino</a>,
42   * and then back to {@code .xmir} and to {@code .class}. As a result,
43   * you obtain optimized Bytecode in the <tt>target/classes/</tt> directory,
44   * which supposedly works faster than before.</p>
45   *
46   * <p>The entire optimization pipeline happens inside Docker container,
47   * which is run from the image specified in the <tt>image</tt> parameter.
48   * The image may either be pulled or built locally. We recommend pulling
49   * it from the Docker Hub with the help of the <tt>pull</tt> goal. Also,
50   * we recommend deleting the image after optimization is done, with the help
51   * of the <tt>rmi</tt> goal.</p>
52   *
53   * @since 0.1.0
54   * @checkstyle CyclomaticComplexityCheck (500 lines)
55   * @checkstyle NPathComplexityCheck (500 lines)
56   * @checkstyle ExecutableStatementCountCheck (500 lines)
57   */
58  @Mojo(name = "optimize", defaultPhase = LifecyclePhase.PROCESS_CLASSES, requiresProject = false)
59  @SuppressWarnings({"PMD.AvoidDuplicateLiterals", "PMD.TooManyFields"})
60  public final class OptimizeMojo extends AbstractMojo {

293     @Override
294     public void exec() throws IOException {
295         if (!this.target.toPath().resolve(this.classes).toFile().exists()
296             && this.skipIfNoClasses) {
297             Logger.info(this, "The directory with classes is absent, skipping");
298             return;
299         }
300         if (this.target.mkdirs()) {
301             Logger.info(this, "Target directory %[file]s created", this.target);
302         } else {
303             Logger.info(this, "Target directory %[file]s already exists", this.target);
304         }
305         final long start = System.currentTimeMillis();
306         if (this.alwaysWithDocker || !new Phino().available()) {
307             this.withDocker();
308         } else {
309             this.withoutDocker();
310         }
311         Logger.info(
312             this,
313             "Bytecode was optimized in '%s' in %[ms]s",
314             this.target,
315             System.currentTimeMillis() - start
316         );
317     }

319     @SuppressWarnings({ "PMD.CognitiveComplexity", "PMD.NPathComplexity", "PMD.NcssCount" })
320     private void withDocker() throws IOException {
321         final String tdir = "/target";
322         final String cdir = "/eo-cache";
323         final Collection<String> command = new LinkedList<>(
324             Arrays.asList(
325                 "run",
326                 "--rm",
327                 "--volume", String.format("%s:%s", this.target, tdir),
328                 "--volume", String.format("%s:%s", this.cache, cdir),
329                 "--env", String.format("TARGET=%s", tdir),
330                 "--env", String.format("EO_CACHE=%s", cdir),
331                 "--env", "WORKDIR=/hone"
332             )
333         );
334         final Path extdir = this.target.toPath().resolve("hone-extra");
335         if (extdir.toFile().mkdirs()) {
336             Logger.info(this, "Directory %[file]s created", extdir);
337         }
338         if (this.extra != null) {
339             this.copyExtras(extdir);
340             command.addAll(
341                 Arrays.asList(
342                     "--env", String.format("EXTRA=%s/hone-extra", tdir)
343                 )
344             );
345         }
346         if (this.eoVersion == null) {
347             Logger.info(this, "EO version is not set, we use the default one");
348         } else {
349             command.addAll(
350                 Arrays.asList(
351                     "--env", String.format("EO_VERSION=%s", this.eoVersion)
352                 )
353             );
354         }
355         if (this.jeoVersion == null) {
356             Logger.info(this, "JEO version is not set, we use the default one");
357         } else {
358             command.addAll(
359                 Arrays.asList(
360                     "--env", String.format("JEO_VERSION=%s", this.jeoVersion)
361                 )
362             );
363         }
364         command.addAll(
365             Arrays.asList(
366                 "--env", String.format("CLASSES=%s", this.classes)
367             )
368         );
369         command.addAll(
370             Arrays.asList(
371                 "--env", String.format("SMALL_STEPS=%s", this.smallSteps)
372             )
373         );
374         command.addAll(
375             Arrays.asList(
376                 "--env", String.format("SKIP_PHINO=%s", this.skipPhino)
377             )
378         );
379         command.addAll(
380             Arrays.asList(
381                 "--env", String.format("VERBOSE=%s", Logger.isDebugEnabled(this))
382             )
383         );
384         command.addAll(
385             Arrays.asList(
386                 "--env", String.format("DEBUG=%s", this.debug)
387             )
388         );
389         command.addAll(
390             Arrays.asList(
391                 "--env", String.format("GREP_IN=%s", this.grepIn)
392             )
393         );
394         command.addAll(
395             Arrays.asList(
396                 "--env", String.format("MAX_DEPTH=%d", this.maxDepth)
397             )
398         );
399         command.addAll(
400             Arrays.asList(
401                 "--env", String.format("MAX_CYCLES=%d", this.maxCycles)
402             )
403         );
404         command.addAll(
405             Arrays.asList(
406                 "--env", String.format("THREADS=%d", this.threads)
407             )
408         );
409         command.addAll(
410             Arrays.asList(
411                 "--env", String.format("TIMEOUT=%d", this.timeout)
412             )
413         );
414         if (this.includes != null && this.includes.length > 0) {
415             command.addAll(
416                 Arrays.asList(
417                     "--env", String.format("INCLUDES=%s", String.join(",", this.includes))
418                 )
419             );
420         }
421         if (this.excludes != null && this.excludes.length > 0) {
422             command.addAll(
423                 Arrays.asList(
424                     "--env", String.format("EXCLUDES=%s", String.join(",", this.excludes))
425                 )
426             );
427         }
428         command.add("--user");
429         command.add(OptimizeMojo.whoami());
430         command.addAll(
431             Arrays.asList(
432                 "--env",
433                 String.format(
434                     "RULES=%s",
435                     this.rulesAsString()
436                 )
437             )
438         );
439         command.add(this.image);
440         this.timings.through(
441             "optimize",
442             () -> new Docker(this.sudo).exec(command)
443         );
444     }

446     private void saveExtra(final Path src, final Path target) throws IOException {
447         final int already = Files.list(target).collect(Collectors.toList()).size();
448         final String name = String.format(
449             "%04d-%s.yml", already,
450             src.getFileName().toString().replaceAll("\\.[a-zA-Z0-9]+$", "")
451         );
452         final Path copy = target.resolve(name);
453         if (copy.toFile().exists()) {
454             throw new IllegalStateException(
455                 String.format(
456                     "Extra rule %s already exists, something is wrong with our algorithm?",
457                     copy
458                 )
459             );
460         }
461         Files.copy(src, copy);
462         Logger.info(
463             this,
464             "Extra rule '%s' copied to %[file]s from %[file]s",
465             name, target, src
466         );
467     }

469     @SuppressWarnings("PMD.CognitiveComplexity")
470     private void copyExtras(final Path extdir) throws IOException {
471         if (this.extra != null) {
472             if (extdir.toFile().mkdirs()) {
473                 Logger.info(this, "Directory %[file]s created", extdir);
474             }
475             for (final String ext : this.extra) {
476                 final Path src = Paths.get(ext);
477                 if (src.toFile().isDirectory()) {
478                     Logger.info(
479                         this,
480                         "Scanning %[file]s for extra rules (%s)...",
481                         src, this.extraExtensions
482                     );
483                     try (Stream<Path> files = Files.list(src)) {
484                         final List<Path> yamls = files
485                             .filter(
486                                 f -> {
487                                     boolean match = false;
488                                     for (final String extn : this.extraExtensions.split(",")) {
489                                         match = match || f.getFileName().toString().endsWith(
490                                             String.format(".%s", extn.trim())
491                                         );
492                                     }
493                                     return match;
494                                 }
495                             )
496                             .sorted()
497                             .collect(Collectors.toList());
498                         for (final Path yaml : yamls) {
499                             this.saveExtra(yaml, extdir);
500                         }
501                     }
502                 } else {
503                     this.saveExtra(src, extdir);
504                 }
505             }
506         }
507     }

532     // @checkstyle CyclomaticComplexityCheck (200 lines)
533     // @checkstyle NPathComplexityCheck (200 lines)
534     @SuppressWarnings("PMD.NPathComplexity")
535     private void withoutDocker() throws IOException {
536         try (Mktemp temp = new Mktemp()) {
537             final String[] files = {
538                 "entry.sh",
539                 "pom.xml",
540                 "normalize.sh",
541             };
542             for (final String file : files) {
543                 new IoChecked<>(
544                     new LengthOf(
545                         new TeeInput(
546                             new ResourceOf(String.format("org/eolang/hone/scaffolding/%s", file)),
547                             new OutputTo(temp.path().resolve(file))
548                         )
549                     )
550                 ).value();
551             }
552             for (final String file : new String[] {"entry.sh", "normalize.sh"}) {
553                 temp.path().resolve(file).toFile().setExecutable(true);
554             }
555             new Rules("*").copyTo(temp.path().resolve("rules"));
556             Jaxec jaxec = new Jaxec(temp.path().resolve("entry.sh").toString())
557                 .withEnv("TARGET", this.target.toString())
558                 .withEnv("DEBUG", Boolean.toString(this.debug))
559                 .withEnv("VERBOSE", Boolean.toString(Logger.isDebugEnabled(this)))
560                 .withEnv("CLASSES", this.classes)
561                 .withEnv("SMALL_STEPS", Boolean.toString(this.smallSteps))
562                 .withEnv("SKIP_PHINO", Boolean.toString(this.skipPhino))
563                 .withEnv("GREP_IN", this.grepIn)
564                 .withEnv("MAX_DEPTH", Integer.toString(this.maxDepth))
565                 .withEnv("MAX_CYCLES", Integer.toString(this.maxCycles))
566                 .withEnv("THREADS", Integer.toString(this.threads))
567                 .withEnv("TIMEOUT", Integer.toString(this.timeout))
568                 .withEnv("RULES", this.rulesAsString());
569             if (this.extra != null && !this.extra.isEmpty()) {
570                 this.copyExtras(temp.path().resolve("hone-extra"));
571                 jaxec = jaxec.withEnv("EXTRA", temp.path().resolve("hone-extra").toString());
572             }
573             if (this.includes != null && this.includes.length > 0) {
574                 jaxec = jaxec.withEnv("INCLUDES", this.localPaths(this.includes));
575             }
576             if (this.excludes != null && this.excludes.length > 0) {
577                 jaxec = jaxec.withEnv("EXCLUDES", this.localPaths(this.excludes));
578             }
579             if (this.cache != null) {
580                 jaxec = jaxec.withEnv("EO_CACHE", this.cache.getAbsolutePath());
581             }
582             if (this.eoVersion == null) {
583                 Logger.info(this, "EO version is not set, we use the default one");
584             } else {
585                 jaxec = jaxec.withEnv("EO_VERSION", this.eoVersion);
586             }
587             if (this.jeoVersion == null) {
588                 Logger.info(this, "JEO version is not set, we use the default one");
589             } else {
590                 jaxec = jaxec.withEnv("JEO_VERSION", this.jeoVersion);
591             }
592             jaxec.exec();
593         }
594     }
