5   package org.eolang.hone;
6   
7   import com.yegor256.Mktmp;
8   import com.yegor256.MktmpResolver;
9   import java.nio.file.Files;
10  import java.nio.file.Path;
11  import org.cactoos.bytes.BytesOf;
12  import org.cactoos.io.ResourceOf;
13  import org.cactoos.text.TextOf;
14  import org.hamcrest.MatcherAssert;
15  import org.hamcrest.Matchers;
16  import org.junit.jupiter.api.Test;
17  import org.junit.jupiter.api.extension.ExtendWith;
18  
19  /**
20   * Test case for {@link Rules}.
21   * @since 0.1.0
22   */
23  @ExtendWith(MktmpResolver.class)
24  final class SummaryTest {
25  
26      @Test
27      void compilesCommonSummaryStatistics(@Mktmp final Path temp) throws Exception {
28          MatcherAssert.assertThat(
29              "report must contain statistics from both modules",
30              new TextOf(new Summary(SummaryTest.modular(temp)).collect()).asString(),
31              Matchers.allOf(
32                  Matchers.containsString(
33                      "/phi/org/eolang/hone/client/Client.phi,/phi-optimized/org/eolang/hone/client/Client.phi,2,4000"
34                  ),
35                  Matchers.containsString(
36                      "/phi/org/eolang/hone/server/Server.phi,/phi-optimized/org/eolang/hone/server/Server.phi,10,3200"
37                  )
38              )
39          );
40      }
41  
42      @Test
43      void skipsModulesWithoutStatistics(@Mktmp final Path temp) throws Exception {
44          final Path dir = SummaryTest.modular(temp);
45          Files.deleteIfExists(dir.resolve("server/hone-statistics.csv"));
46          Files.deleteIfExists(dir.resolve("client/hone-statistics.csv"));
47          MatcherAssert.assertThat(
48              "report shouldn't be generated if no statistics found",
49              new Summary(dir).collect().toFile().exists(),
50              Matchers.is(false)
51          );
52      }
53  
54      private static Path modular(final Path root) throws Exception {
55          Files.createDirectories(root.resolve("server"));
56          Files.createDirectories(root.resolve("client"));
57          Files.write(
58              root.resolve("server/hone-statistics.csv"),
59              new BytesOf(new ResourceOf("csv/hone-statistics-server.csv")).asBytes()
60          );
61          Files.write(
62              root.resolve("client/hone-statistics.csv"),
63              new BytesOf(new ResourceOf("csv/hone-statistics-client.csv")).asBytes()
64          );
65          return root;
66      }
67  }
