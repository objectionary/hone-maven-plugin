5   package org.eolang.hone;
6   
7   import com.jcabi.log.Logger;
8   import java.io.IOException;
9   import java.nio.file.Files;
10  import java.nio.file.Path;
11  import org.apache.maven.plugins.annotations.Mojo;
12  
13  /**
14   * Build summary statistics.
15   *
16   * <p>This goal tries to find all the statistics in child modules and
17   * build a single summary report.</p>
18   *
19   * @since 0.1.0
20   */
21  @Mojo(
22      name = "summary",
23      aggregator = true
24  )
25  public final class SummaryMojo extends AbstractMojo {
26  
27      @Override
28      public void exec() throws IOException {
29          Logger.info(this, "Collecting summary build statistics...");
30          final Path stats = new Summary(
31              this.basedir.toPath(),
32              Files.createDirectories(this.target.toPath())
33          ).collect();
34          if (stats.toFile().exists()) {
35              final CSV csv = new CSV(stats);
36              Logger.info(
37                  this,
38                  "Optimized %d/%d files",
39                  csv.count("Changed", v -> Integer.parseInt(v) > 0),
40                  csv.size()
41              );
42          }
43          Logger.info(this, "Summary build statistics collected successfully!");
44      }
45  }
