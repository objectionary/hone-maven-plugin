5   package org.eolang.hone;
6   
7   import java.nio.file.Files;
8   import java.nio.file.Path;
9   import org.hamcrest.MatcherAssert;
10  import org.hamcrest.Matchers;
11  import org.junit.jupiter.api.Test;
12  
13  /**
14   * Test case for {@link Timings}.
15   *
16   * @since 0.1.0
17   */
18  final class TimingsTest {
19  
20      @Test
21      void savesTime() throws Exception {
22          try (Mktemp temp = new Mktemp()) {
23              final Path file = temp.path().resolve("foo.csv");
24              final Timings timings = new Timings(file);
25              timings.through("foo", () -> { });
26              MatcherAssert.assertThat(
27                  "file must be written",
28                  file.toFile().exists(),
29                  Matchers.is(true)
30              );
31              timings.through("bar", () -> { });
32              MatcherAssert.assertThat(
33                  "file must have two lines",
34                  new String(Files.readAllBytes(file)),
35                  Matchers.allOf(
36                      Matchers.containsString("foo,"),
37                      Matchers.containsString("\nbar,")
38                  )
39              );
40          }
41      }
42  }
