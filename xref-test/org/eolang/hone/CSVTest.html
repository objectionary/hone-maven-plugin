5   package org.eolang.hone;
6   
7   import com.yegor256.Mktmp;
8   import com.yegor256.MktmpResolver;
9   import java.nio.charset.StandardCharsets;
10  import java.nio.file.Files;
11  import java.nio.file.Path;
12  import org.hamcrest.MatcherAssert;
13  import org.hamcrest.Matchers;
14  import org.junit.jupiter.api.Test;
15  import org.junit.jupiter.api.extension.ExtendWith;
16  
17  /**
18   * Test case for {@link CSV}.
19   * @since 0.1.0
20   * @checkstyle AbbreviationAsWordInNameCheck (3 lines)
21   */
22  @ExtendWith(MktmpResolver.class)
23  final class CSVTest {
24  
25      @Test
26      void countsRowsMatchingCondition(@Mktmp final Path temp) throws Exception {
27          final Path path = temp.resolve("test.csv");
28          Files.write(
29              path,
30              String.join(
31                  System.lineSeparator(),
32                  "ID,Before,After,Changed,LinesPerSec",
33                  "1/3,a.phi,b.phi,5,1000",
34                  "2/3,c.phi,d.phi,0,0",
35                  "3/3,e.phi,f.phi,3,800",
36                  ""
37              ).getBytes(StandardCharsets.UTF_8)
38          );
39          MatcherAssert.assertThat(
40              "two files have Changed > 0",
41              new CSV(path).count("Changed", v -> Integer.parseInt(v) > 0),
42              Matchers.is(2)
43          );
44      }
45  
46      @Test
47      void countsZeroWhenNoRowsMatch(@Mktmp final Path temp) throws Exception {
48          final Path path = temp.resolve("test.csv");
49          Files.write(
50              path,
51              String.join(
52                  System.lineSeparator(),
53                  "ID,Before,After,Changed,LinesPerSec",
54                  "1/2,a.phi,b.phi,0,0",
55                  "2/2,c.phi,d.phi,0,0",
56                  ""
57              ).getBytes(StandardCharsets.UTF_8)
58          );
59          MatcherAssert.assertThat(
60              "no files have Changed > 0",
61              new CSV(path).count("Changed", v -> Integer.parseInt(v) > 0),
62              Matchers.is(0)
63          );
64      }
65  
66      @Test
67      void parsesHeaderOnlyCsv(@Mktmp final Path temp) throws Exception {
68          final Path path = temp.resolve("test.csv");
69          Files.write(
70              path,
71              String.join(
72                  System.lineSeparator(),
73                  "ID,Before,After,Changed,LinesPerSec",
74                  ""
75              ).getBytes(StandardCharsets.UTF_8)
76          );
77          MatcherAssert.assertThat(
78              "header-only CSV has zero data rows",
79              new CSV(path).size(),
80              Matchers.is(0)
81          );
82      }
83  
84      @Test
85      void countsZeroOnHeaderOnlyCsv(@Mktmp final Path temp) throws Exception {
86          final Path path = temp.resolve("test.csv");
87          Files.write(
88              path,
89              String.join(
90                  System.lineSeparator(),
91                  "ID,Before,After,Changed,LinesPerSec",
92                  ""
93              ).getBytes(StandardCharsets.UTF_8)
94          );
95          MatcherAssert.assertThat(
96              "header-only CSV reports zero matches",
97              new CSV(path).count("Changed", v -> Integer.parseInt(v) > 0),
98              Matchers.is(0)
99          );
100     }
101 
102     @Test
103     void parsesCompletelyEmptyCsv(@Mktmp final Path temp) throws Exception {
104         final Path path = temp.resolve("empty.csv");
105         Files.write(path, new byte[0]);
106         MatcherAssert.assertThat(
107             "empty CSV has zero data rows without throwing IndexOutOfBoundsException",
108             new CSV(path).size(),
109             Matchers.is(0)
110         );
111     }
112 
113     @Test
114     void countsZeroOnCompletelyEmptyCsv(@Mktmp final Path temp) throws Exception {
115         final Path path = temp.resolve("empty.csv");
116         Files.write(path, new byte[0]);
117         MatcherAssert.assertThat(
118             "empty CSV reports zero matches without throwing IndexOutOfBoundsException",
119             new CSV(path).count("Changed", v -> Integer.parseInt(v) > 0),
120             Matchers.is(0)
121         );
122     }
123 }
