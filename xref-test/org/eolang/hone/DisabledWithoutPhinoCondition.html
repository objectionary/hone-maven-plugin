5   package org.eolang.hone;
6   
7   import java.io.IOException;
8   import org.cactoos.io.ResourceOf;
9   import org.cactoos.text.IoCheckedText;
10  import org.cactoos.text.TextOf;
11  import org.junit.jupiter.api.extension.ConditionEvaluationResult;
12  import org.junit.jupiter.api.extension.ExecutionCondition;
13  import org.junit.jupiter.api.extension.ExtensionContext;
14  
15  /**
16   * This annotation disables JUnit tests when phino is
17   * not runnable in command line.
18   * @since 0.1.0
19   */
20  public final class DisabledWithoutPhinoCondition implements ExecutionCondition {
21  
22      @Override
23      public ConditionEvaluationResult evaluateExecutionCondition(final ExtensionContext ctx) {
24          final String expected;
25          try {
26              expected = new IoCheckedText(
27                  new TextOf(
28                      new ResourceOf("org/eolang/hone/default-phino-version.txt")
29                  )
30              ).asString().trim();
31          } catch (final IOException ex) {
32              throw new IllegalStateException(
33                  "Cannot read the pinned phino version from default-phino-version.txt",
34                  ex
35              );
36          }
37          final ConditionEvaluationResult result;
38          if (new Phino().available(expected)) {
39              result = ConditionEvaluationResult.enabled("Phino is available");
40          } else {
41              result = ConditionEvaluationResult.disabled("Phino is not available");
42          }
43          return result;
44      }
45  }
