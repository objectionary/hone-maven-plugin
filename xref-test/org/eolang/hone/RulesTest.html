5   package org.eolang.hone;
6   
7   import com.yegor256.Mktmp;
8   import com.yegor256.MktmpResolver;
9   import java.io.IOException;
10  import java.nio.file.Path;
11  import org.hamcrest.MatcherAssert;
12  import org.hamcrest.Matchers;
13  import org.junit.jupiter.api.Test;
14  import org.junit.jupiter.api.extension.ExtendWith;
15  
16  /**
17   * Test case for {@link Rules}.
18   *
19   * @since 0.1.0
20   */
21  @ExtendWith(MktmpResolver.class)
22  final class RulesTest {
23  
24      @Test
25      void filtersAndSaves(@Mktmp final Path temp) throws Exception {
26          final Rules rules = new Rules("n*,aaa*,*{,!f*");
27          rules.copyTo(temp.resolve("a/b/c"));
28          MatcherAssert.assertThat(
29              String.format("file must be written, because of %s", rules),
30              temp.resolve("a/b/c/none.yml").toFile().exists(),
31              Matchers.is(true)
32          );
33      }
34  
35      @Test
36      void skipsSome(@Mktmp final Path temp) throws Exception {
37          final Rules rules = new Rules("!none,33*");
38          rules.copyTo(temp.resolve("a/b/c"));
39          MatcherAssert.assertThat(
40              String.format("file must be written, because of %s", rules),
41              temp.resolve("a/b/c/33-to-42.yml").toFile().exists(),
42              Matchers.is(true)
43          );
44          MatcherAssert.assertThat(
45              String.format("file must be absent, because of %s", rules),
46              temp.resolve("a/b/c/none.yml").toFile().exists(),
47              Matchers.is(false)
48          );
49      }
50  
51      @Test
52      void discoversOneRuleFromClasspathWithoutSuffix() {
53          MatcherAssert.assertThat(
54              "Should discover none from classpath (without YML suffix)",
55              new Rules("none").yamls(),
56              Matchers.iterableWithSize(1)
57          );
58      }
59  
60      @Test
61      void discoversNothingWithSuffix() {
62          MatcherAssert.assertThat(
63              "Should NOT discover none.yml from classpath",
64              new Rules("none.yml").yamls(),
65              Matchers.emptyIterable()
66          );
67      }
68  
69      @Test
70      void discoversRulesFromClasspath() {
71          MatcherAssert.assertThat(
72              "Should discover none.yml from classpath",
73              new Rules("*").yamls(),
74              Matchers.hasItems(
75                  "none.yml",
76                  "streams/701-static-lambda-to-invokedynamic.phr"
77              )
78          );
79      }
80  
81      @Test
82      void copiesAllRulesFromClasspath(@Mktmp final Path temp) throws IOException {
83          new Rules("*").copyTo(temp.resolve("copies"));
84          MatcherAssert.assertThat(
85              "Should copy .phr rules too",
86              temp.resolve(
87                  "copies/streams/701-static-lambda-to-invokedynamic.phr"
88              ).toFile().exists(),
89              Matchers.is(true)
90          );
91      }
92  
93      @Test
94      void yamlsReturnsSortedList() {
95          final Iterable<String> yamls = new Rules("*").yamls();
96          String previous = null;
97          for (final String yaml : yamls) {
98              if (previous != null) {
99                  MatcherAssert.assertThat(
100                     String.format(
101                         "yamls() is not sorted: '%s' > '%s'",
102                         previous, yaml
103                     ),
104                     previous.compareTo(yaml),
105                     Matchers.lessThanOrEqualTo(0)
106                 );
107             }
108             previous = yaml;
109         }
110     }
111 
112 }
