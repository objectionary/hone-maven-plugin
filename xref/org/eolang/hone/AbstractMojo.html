5   package org.eolang.hone;
6   
7   import com.jcabi.log.Logger;
8   import java.io.File;
9   import java.io.IOException;
10  import org.apache.maven.plugin.MojoExecutionException;
11  import org.apache.maven.plugins.annotations.Parameter;
12  import org.slf4j.impl.StaticLoggerBinder;
13  
14  /**
15   * Abstract base class for all Hone Maven plugin goals.
16   *
17   * <p>This class provides common functionality shared by all plugin goals,
18   * including Docker image configuration, sudo support, execution timing,
19   * and skip functionality. Unfortunately, this is the best we can do
20   * in Maven Plugin API design (this inheritance-based approach is ugly,
21   * but Maven forces us to use it).</p>
22   *
23   * @since 0.1.0
24   * @checkstyle VisibilityModifierCheck (500 lines)
25   */
26  abstract class AbstractMojo extends org.apache.maven.plugin.AbstractMojo {
27  
28      /**
29       * The "target/" directory of Maven project.
30       *
31       * @since 0.1.0
32       */
33      @Parameter(
34          property = "hone.target",
35          defaultValue = "${project.build.directory}"
36      )
37      protected File target;
38  
39      /**
40       * Timings tracker for performance measurements.
41       *
42       * @since 0.1.0
43       */
44      protected Timings timings;
45  
46      /**
47       * Docker image to use.
48       *
49       * @since 0.1.0
50       */
51      @Parameter(property = "hone.image", defaultValue = "yegor256/hone:0.12.0")
52      protected String image;
53  
54      /**
55       * Whether to use "sudo" when executing Docker commands.
56       *
57       * @since 0.1.0
58       */
59      @Parameter(property = "hone.sudo", defaultValue = "false")
60      protected boolean sudo;
61  
62      /**
63       * Skip the execution, if set to TRUE.
64       *
65       * @since 0.1.0
66       */
67      @Parameter(property = "hone.skip", defaultValue = "false")
68      private boolean skip;
69  
70      @Override
71      public final void execute() throws MojoExecutionException {
72          StaticLoggerBinder.getSingleton().setMavenLog(this.getLog());
73          if (this.skip) {
74              Logger.info(this, "Execution skipped due to hone.skip=true");
75              return;
76          }
77          this.timings = new Timings(this.target.toPath().resolve("hone-timings.csv"));
78          try {
79              this.exec();
80          } catch (final IOException ex) {
81              throw new MojoExecutionException(ex);
82          }
83      }
84  
85      /**
86       * Execute the specific goal implementation.
87       * @throws IOException If execution fails
88       */
89      abstract void exec() throws IOException;
90  }
