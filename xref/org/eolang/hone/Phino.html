5   package org.eolang.hone;
6   
7   import com.jcabi.log.Logger;
8   import com.yegor256.Jaxec;
9   import com.yegor256.Result;
10  import java.io.IOException;
11  
12  /**
13   * An abstraction of Phino in command line.
14   *
15   * @since 0.17.0
16   */
17  final class Phino {
18  
19      /**
20       * Is it available?
21       * @param expected This is the expected version
22       * @return TRUE if available
23       */
24      public boolean available(final String expected) {
25          boolean available = false;
26          try {
27              final Result result = new Jaxec("phino", "--version").withCheck(false).execUnsafe();
28              if (result.code() == 0) {
29                  final String version = result.stdout().trim();
30                  if (version.equals(expected)) {
31                      available = true;
32                      Logger.info(
33                          this,
34                          String.format(
35                              "The 'phino' executable found (%s), no need to use Docker",
36                              version
37                          )
38                      );
39                  } else {
40                      Logger.info(
41                          this,
42                          "The 'phino' executable is found, but its version (%s) is not equal to the expected one (%s)",
43                          version, expected
44                      );
45                  }
46              } else {
47                  Logger.info(
48                      this,
49                      "The 'phino' executable is found, but it doesn't work, we must use Docker"
50                  );
51              }
52          } catch (final IOException ex) {
53              Logger.info(
54                  this,
55                  String.format(
56                      "The 'phino' executable not found, we must use Docker: %s",
57                      ex.getMessage()
58                  )
59              );
60          }
61          return available;
62      }
63  
64  }
