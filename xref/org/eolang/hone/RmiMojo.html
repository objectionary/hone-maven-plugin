5   package org.eolang.hone;
6   
7   import com.jcabi.log.Logger;
8   import java.io.IOException;
9   import org.apache.maven.plugins.annotations.LifecyclePhase;
10  import org.apache.maven.plugins.annotations.Mojo;
11  
12  /**
13   * Remove Docker image.
14   *
15   * <p>This goal deletes Docker image from your machine, in order to
16   * save space and simply clean up after the optimization step. You
17   * may not use this goal at all, but we recommend to use it.</p>
18   *
19   * @since 0.1.0
20   */
21  @Mojo(name = "rmi", defaultPhase = LifecyclePhase.PROCESS_CLASSES, requiresProject = false)
22  public final class RmiMojo extends AbstractMojo {
23  
24      @Override
25      public void exec() throws IOException {
26          new Docker(this.sudo).exec(
27              "rmi",
28              "--no-prune",
29              this.image
30          );
31          Logger.info(this, "Docker image '%s' was removed", this.image);
32      }
33  }
