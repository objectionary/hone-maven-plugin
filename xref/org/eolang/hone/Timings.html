24  package org.eolang.hone;
25  
26  import com.jcabi.log.Logger;
27  import java.io.File;
28  import java.io.IOException;
29  import java.nio.charset.StandardCharsets;
30  import java.nio.file.Files;
31  import java.nio.file.Path;
32  import java.nio.file.StandardOpenOption;
33  
34  /**
35   * Timings.
36   *
37   * @since 0.1.0
38   */
39  final class Timings {
40  
41      /**
42       * Path of the file.
43       */
44      private final Path path;
45  
46      /**
47       * Ctor.
48       * @param file Location of the CSV file to modify
49       */
50      Timings(final Path file) {
51          this.path = file;
52      }
53  
54      /**
55       * Run and record.
56       * @param name Name of the action
57       * @param action The action
58       * @throws IOException If fails
59       */
60      public void through(final String name, final Timings.Action action) throws IOException {
61          final long start = System.currentTimeMillis();
62          try {
63              action.exec();
64          } finally {
65              final File dir = this.path.toFile().getParentFile();
66              if (dir.mkdirs()) {
67                  Logger.debug(this, "Directory created: %[file]s", dir);
68              }
69              Files.write(
70                  this.path,
71                  String.format(
72                      "%s,%d\n", name, System.currentTimeMillis() - start
73                  ).getBytes(StandardCharsets.UTF_8),
74                  StandardOpenOption.APPEND, StandardOpenOption.CREATE
75              );
76          }
77      }
78  
79      /**
80       * What to do.
81       *
82       * @since 0.1.0
83       */
84      public interface Action {
85          /**
86           * Exec it.
87           * @throws IOException If fails
88           */
89          void exec() throws IOException;
90      }
91  
92  }
